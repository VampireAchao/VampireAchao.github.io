```javascript
# JavaScript模板字符串使用
var name = "achao";
console.log(`我是${name}`);
var number = 0;
console.log(`今年${(number + 38) / 2}`);
var word = "模板字符串";
console.log(`${name}说：\'${word}\'那不是一般的香\n${rubenSaid()}`);

function rubenSaid() {
    return `ruben说：${name}说的对!`;
}
```
